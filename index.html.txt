// ========== SOUND EFFECTS (using Web Audio API - no files needed) ==========
const AudioCtx = window.AudioContext || window.webkitAudioContext;
let audioCtx = null;
function initAudio() { if (!audioCtx) audioCtx = new AudioCtx(); }

function playTone(frequencies, duration, type = 'sine') {
  initAudio();
  const now = audioCtx.currentTime;
  frequencies.forEach((f, i) => {
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = type;
    osc.frequency.value = f.freq;
    gain.gain.setValueAtTime(0.15, now + f.start);
    gain.gain.exponentialRampToValueAtTime(0.001, now + f.start + f.dur);
    osc.connect(gain); gain.connect(audioCtx.destination);
    osc.start(now + f.start); osc.stop(now + f.start + f.dur);
  });
}

const SFX = {
  correct: () => playTone([
    {freq: 523, start: 0, dur: 0.12},
    {freq: 659, start: 0.1, dur: 0.12},
    {freq: 784, start: 0.2, dur: 0.2}
  ]),
  wrong: () => playTone([
    {freq: 220, start: 0, dur: 0.2},
    {freq: 180, start: 0.15, dur: 0.3}
  ], 0, 'sawtooth'),
  finish: () => playTone([
    {freq: 523, start: 0, dur: 0.15},
    {freq: 659, start: 0.15, dur: 0.15},
    {freq: 784, start: 0.3, dur: 0.15},
    {freq: 1047, start: 0.45, dur: 0.4}
  ])
};

// ========== GAME STATE ==========
const Game = {
  players: [],
  currentPlayerIndex: 0,
  categories: [],
  board: {},
  activeQuestion: null,
  totalQuestions: 40,
  answered: 0,
  stealLog: [],
  usedQuestionIndices: {}, // {category: Set of used question array indices}

  init(names) {
    this.players = names.map(n => ({
      name: n,
      score: 0,
      steals: 5,
      stealCount: 0,
      tools: { hint: true, choices: true, two: true, replace: true }
    }));
    this.currentPlayerIndex = 0;
    this.answered = 0;
    this.stealLog = [];
    this.usedQuestionIndices = {};
    const all = Object.keys(CATEGORIES);
    this.categories = [...all].sort(() => Math.random() - 0.5).slice(0, 4);
    this.buildBoard();
  },

  buildBoard() {
    this.board = {};
    const points = [100,200,300,400,500,600,700,800,900,1000];
    this.categories.forEach(cat => {
      this.usedQuestionIndices[cat] = new Set();
      const pool = CATEGORIES[cat];
      this.board[cat] = [];
      for (let diffIdx = 0; diffIdx < 10; diffIdx++) {
        // Filter pool by difficulty (points bracket)
        const candidates = pool.filter((q, i) => q.difficulty === diffIdx && !this.usedQuestionIndices[cat].has(i));
        // Pick random one
        let chosen;
        if (candidates.length > 0) {
          chosen = candidates[Math.floor(Math.random() * candidates.length)];
        } else {
          // Fallback: any unused question
          const fallback = pool.filter((q, i) => !this.usedQuestionIndices[cat].has(i));
          chosen = fallback[Math.floor(Math.random() * fallback.length)] || pool[0];
        }
        const origIdx = pool.indexOf(chosen);
        this.usedQuestionIndices[cat].add(origIdx);
        this.board[cat].push({
          points: points[diffIdx],
          difficulty: diffIdx,
          q: chosen.q,
          a: chosen.a,
          hint: chosen.hint || "No hint available.",
          choices: chosen.choices || [chosen.a, "Option B", "Option C", "Option D"],
          used: false
        });
      }
    });
  },

  currentPlayer() { return this.players[this.currentPlayerIndex]; },

  selectQuestion(cat, idx) {
    const t = this.board[cat][idx];
    if (t.used) return null;
    this.activeQuestion = { category: cat, index: idx, ...t };
    return this.activeQuestion;
  },

  replaceActiveQuestion() {
    // Pull a fresh unused question at the same difficulty from the pool
    const cat = this.activeQuestion.category;
    const diff = this.activeQuestion.difficulty;
    const pool = CATEGORIES[cat];
    const candidates = pool.filter((q, i) => q.difficulty === diff && !this.usedQuestionIndices[cat].has(i));
    if (candidates.length === 0) return false;
    const chosen = candidates[Math.floor(Math.random() * candidates.length)];
    const origIdx = pool.indexOf(chosen);
    this.usedQuestionIndices[cat].add(origIdx);
    const boardIdx = this.activeQuestion.index;
    this.board[cat][boardIdx].q = chosen.q;
    this.board[cat][boardIdx].a = chosen.a;
    this.board[cat][boardIdx].hint = chosen.hint || "No hint available.";
    this.board[cat][boardIdx].choices = chosen.choices || [chosen.a,"B","C","D"];
    this.activeQuestion.q = chosen.q;
    this.activeQuestion.a = chosen.a;
    this.activeQuestion.hint = this.board[cat][boardIdx].hint;
    this.activeQuestion.choices = this.board[cat][boardIdx].choices;
    return true;
  },

  resolveQuestion(correct, scorerIndex = null) {
    const { category, index, points } = this.activeQuestion;
    this.board[category][index].used = true;
    const scorer = (scorerIndex !== null) ? this.players[scorerIndex] : this.currentPlayer();
    if (correct) scorer.score += points;
    this.answered++;
    this.activeQuestion = null;
    this.currentPlayerIndex = (this.currentPlayerIndex + 1) % this.players.length;
  },

  logSteal(stealerIdx, originalIdx, category, points, correct) {
    this.stealLog.push({
      stealer: this.players[stealerIdx].name,
      from: this.players[originalIdx].name,
      category, points, correct,
      turn: this.answered + 1
    });
    this.players[stealerIdx].steals--;
    this.players[stealerIdx].stealCount++;
  },

  isGameOver() { return this.answered >= this.totalQuestions; },
  finalRanking() { return [...this.players].sort((a,b) => b.score - a.score); }
};

// ===== STEAL PHASE =====
function showStealPhase() {
  const box = $("steal-buttons"); box.innerHTML = "";
  let anyAvailable = false;
  Game.players.forEach((p, idx) => {
    if (idx === Game.currentPlayerIndex) return; // can't steal own
    const btn = document.createElement("button");
    btn.className = "steal-btn";
    btn.textContent = p.name + " (" + p.steals + " left)";
    btn.disabled = p.steals <= 0;
    if (p.steals > 0) anyAvailable = true;
    btn.addEventListener("click", () => attemptSteal(idx));
    box.appendChild(btn);
  });
  if (!anyAvailable) {
    // No one can steal - just resolve as wrong
    resolveAndAdvance(false, Game.currentPlayerIndex);
    return;
  }
  $("correct-btn").classList.add("hidden");
  $("wrong-btn").classList.add("hidden");
  $("steal-section").classList.remove("hidden");
}

function attemptSteal(stealerIdx) {
  const originalIdx = Game.currentPlayerIndex;
  const cat = Game.activeQuestion.category;
  const points = Game.activeQuestion.points;
  const wantCorrect = confirm(
    Game.players[stealerIdx].name + " is stealing!\n\nDid they answer correctly?\n\nOK = Correct (they get " + points + " pts)\nCancel = Wrong (steal used, no points)"
  );
  Game.logSteal(stealerIdx, originalIdx, cat, points, wantCorrect);
  if (wantCorrect) {
    SFX.correct();
    resolveAndAdvance(true, stealerIdx);
  } else {
    SFX.wrong();
    resolveAndAdvance(false, originalIdx);
  }
}

$("skip-steal-btn").addEventListener("click", () => {
  resolveAndAdvance(false, Game.currentPlayerIndex);
});

function resolveAndAdvance(correct, scorerIdx) {
  Game.resolveQuestion(correct, scorerIdx);
  $("question-modal").classList.add("hidden");
  renderBoard(); updateHUD();
  if (Game.isGameOver()) endGame();
}

// ===== END GAME =====
$("end-game-btn").addEventListener("click", () => {
  if (confirm("End game now?")) endGame();
});

function endGame() {
  SFX.finish();
  const ranked = Game.finalRanking();
  const c = $("final-scores"); c.innerHTML = "";
  ranked.forEach((p, i) => {
    const row = document.createElement("div");
    row.className = "final-row" + (i === 0 ? " winner" : "");
    row.innerHTML = "<span>" + (i === 0 ? "🏆 " : "#" + (i+1) + " ") + p.name +
      "</span><span>" + p.score + " pts (" + p.stealCount + " steals used)</span>";
    c.appendChild(row);
  });
  const log = $("steal-log"); log.innerHTML = "";
  if (Game.stealLog.length === 0) {
    log.innerHTML = "<div>No steals happened this game.</div>";
  } else {
    Game.stealLog.forEach(s => {
      const d = document.createElement("div");
      d.textContent = "Turn " + s.turn + " · " + s.stealer + " stole from " + s.from +
        " (" + s.category + ", " + s.points + " pts) → " + (s.correct ? "✔ Won " + s.points + "pts" : "✘ Missed");
      log.appendChild(d);
    });
  }
  showScreen("end-screen");
